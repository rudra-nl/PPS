#24
Find roots of quadratic equation 
▼



  
C Program

  
Calculate discriminant (b*b - 4*a*c).


  
#include <stdio.h>
#include <math.h>

int main() {
    float a, b, c, discriminant, root1, root2;
    
    printf("Enter coefficients a, b and c: ");
    scanf("%f %f %f", &a, &b, &c);
    
    discriminant = b * b - 4 * a * c;
    
    if (discriminant > 0) {
        root1 = (-b + sqrt(discriminant)) / (2 * a);
        root2 = (-b - sqrt(discriminant)) / (2 * a);
        printf("Roots are real and distinct: %.2f and %.2f\n", root1, root2);
    } else if (discriminant == 0) {
        root1 = root2 = -b / (2 * a);
        printf("Roots are real and equal: %.2f and %.2f\n", root1, root2);
    } else {
        float realPart = -b / (2 * a);
        float imagPart = sqrt(-discriminant) / (2 * a);
        printf("Roots are complex: %.2f+%.2fi and %.2f-%.2fi\n", realPart, imagPart, realPart, imagPart);
    }
    
    return 0;
}

void processArrays(int *a, int n1, int *b, int n2, int *res, int *n3) {
    int total = n1 + n2, temp;
    
    // Merge
    for (int i = 0; i < n1; i++) *(res + i) = *(a + i);
    for (int i = 0; i < n2; i++) *(res + n1 + i) = *(b + i);
    
    // Sort
    for (int i = 0; i < total - 1; i++) {
        for (int j = 0; j < total - i - 1; j++) {
            if (*(res + j) > *(res + j + 1)) {
                temp = *(res + j);
                *(res + j) = *(res + j + 1);
                *(res + j + 1) = temp;
            }
        }
    }
    
    // Eliminate Duplicates in place
    int k = 0;
    for (int i = 0; i < total; i++) {
        if (i == 0 || *(res + i) != *(res + i - 1)) {
            *(res + k++) = *(res + i);
        }
    }
    *n3 = k; // Update final size
}

int partition(int arr[], int low, int high) {
    int pivot = arr[high];
    int i = (low - 1);
    
    for (int j = low; j <= high - 1; j++) {
        if (arr[j] < pivot) {
            i++;
            swap(&arr[i], &arr[j]);
        }
    }
    swap(&arr[i + 1], &arr[high]);
    return (i + 1);
}

    }
}

void merge(int arr[], int l, int m, int r) {
    int i, j, k;
    int n1 = m - l + 1;
    int n2 = r - m;
    
    int L[n1], R[n2];
    
    for (i = 0; i < n1; i++) L[i] = arr[l + i];
    for (j = 0; j < n2; j++) R[j] = arr[m + 1 + j];
    
    i = 0; j = 0; k = l;
    while (i < n1 && j < n2) {
        if (L[i] <= R[j]) {
            arr[k++] = L[i++];
        } else {
            arr[k++] = R[j++];
        }
    }
    
    while (i < n1) arr[k++] = L[i++];
    while (j < n2) arr[k++] = R[j++];
}

}

    for (int gap = n / 2; gap > 0; gap /= 2) {
        for (int i = gap; i < n; i += 1) {
            int temp = arr[i];
